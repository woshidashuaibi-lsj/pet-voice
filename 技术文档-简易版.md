## 五、核心功能实现

### 5.1 首页录音（index/index.js）

```javascript
const recorderManager = wx.getRecorderManager()

Page({
  data: {
    isRecording: false,
    duration: 0,
    petName: '我的宠物',
    petType: 'cat',  // cat | dog
  },

  onLoad() {
    // 从云数据库加载宠物信息
    this.loadPetInfo()

    // 录音结束回调
    recorderManager.onStop((res) => {
      this.uploadAndAnalyze(res.tempFilePath)
    })
  },

  // 长按开始录音
  onRecordStart() {
    wx.authorize({ scope: 'scope.record' }).then(() => {
      this.setData({ isRecording: true, duration: 0 })
      this._timer = setInterval(() => {
        const d = this.data.duration + 1
        this.setData({ duration: d })
        if (d >= 15) this.onRecordEnd() // 最长 15 秒自动停
      }, 1000)
      recorderManager.start({
        format: 'mp3',
        sampleRate: 16000,
        numberOfChannels: 1,
        duration: 15000,
      })
    }).catch(() => {
      wx.showToast({ title: '请授权麦克风', icon: 'none' })
    })
  },

  // 松手停止录音
  onRecordEnd() {
    clearInterval(this._timer)
    if (this.data.duration < 3) {
      wx.showToast({ title: '请至少录音 3 秒', icon: 'none' })
      recorderManager.stop()
      this.setData({ isRecording: false })
      return
    }
    recorderManager.stop()
    this.setData({ isRecording: false })
  },

  // 上传音频 → 调云函数分析
  async uploadAndAnalyze(tempFilePath) {
    wx.showLoading({ title: '翻译中...', mask: true })

    try {
      // 1. 上传到云存储
      const cloudPath = `audio/${Date.now()}.mp3`
      const uploadRes = await wx.cloud.uploadFile({ cloudPath, filePath: tempFilePath })

      // 2. 调用云函数
      const result = await wx.cloud.callFunction({
        name: 'analyzeAudio',
        data: {
          fileID: uploadRes.fileID,
          petType: this.data.petType,
          petName: this.data.petName,
        }
      })

      wx.hideLoading()

      // 3. 跳转结果页
      const data = result.result
      wx.navigateTo({
        url: `/pages/result/index?data=${encodeURIComponent(JSON.stringify(data))}`
      })
    } catch (e) {
      wx.hideLoading()
      wx.showToast({ title: '分析失败，请重试', icon: 'none' })
      console.error(e)
    }
  },

  async loadPetInfo() {
    const db = wx.cloud.database()
    const res = await db.collection('pets')
      .where({ _openid: '{openid}' })
      .limit(1)
      .get()
    if (res.data.length > 0) {
      const pet = res.data[0]
      this.setData({ petName: pet.name, petType: pet.type })
    }
  }
})
```

### 5.5 核心云函数（analyzeAudio/index.js）

```javascript
// cloudfunctions/analyzeAudio/index.js
const cloud = require('wx-server-sdk')
const https = require('https')

cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })

exports.main = async (event) => {
  const { fileID, petType, petName } = event

  try {
    // 1. 从云存储下载音频（转 base64）
    const fileRes = await cloud.downloadFile({ fileID })
    const audioBase64 = fileRes.fileContent.toString('base64')

    // 2. 调用通义千问多模态 API（音频理解）
    const aiResult = await callQwenAudio(audioBase64, petType, petName)

    // 3. 分析完成后删除云存储中的音频（节省空间）
    await cloud.deleteFile({ fileList: [fileID] })

    return {
      success: true,
      petName,
      ...aiResult,
    }
  } catch (e) {
    console.error('分析失败:', e)
    return { success: false, error: e.message }
  }
}

// 调用通义千问 API
function callQwenAudio(audioBase64, petType, petName) {
  return new Promise((resolve, reject) => {
    const petTypeText = petType === 'cat' ? '猫' : '狗'

    const prompt = `
你是一个专业的宠物行为分析师和宠物翻译官。
我刚才录了一段我家${petTypeText}（名字叫${petName}）的叫声音频。

请你：
1. 判断它现在的情绪状态，从以下选项中选一个：
   happy（开心）/ anxious（焦虑）/ angry（愤怒）/ fearful（恐惧）
   hungry（饥饿）/ playful（想玩）/ bored（无聊）/ uncomfortable（不舒服）
2. 给出置信度（0-100的整数）
3. 用第一人称（${petName}的视角），写一段 50-60 字的可爱解读文案
4. 给主人一条简短的互动建议（15字以内）
5. 如果情绪是 uncomfortable，healthAlert 设为 true，否则设为 false

请严格按以下 JSON 格式返回，不要有其他内容：
{
  "emotion": "xxx",
  "confidence": 85,
  "description": "主人～...",
  "suggestion": "xxx",
  "healthAlert": false
}
`

    const body = JSON.stringify({
      model: 'qwen-audio-turbo',
      input: {
        messages: [
          {
            role: 'user',
            content: [
              { audio: `data:audio/mp3;base64,${audioBase64}` },
              { text: prompt }
            ]
          }
        ]
      }
    })

    const options = {
      hostname: 'dashscope.aliyuncs.com',
      path: '/api/v1/services/aigc/multimodal-generation/generation',
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${process.env.QWEN_API_KEY}`,  // 在云函数环境变量中设置
      }
    }

    const req = https.request(options, (res) => {
      let data = ''
      res.on('data', chunk => { data += chunk })
      res.on('end', () => {
        try {
          const response = JSON.parse(data)
          const text = response.output.choices[0].message.content[0].text
          // 提取 JSON
          const jsonMatch = text.match(/\{[\s\S]*\}/)
          if (jsonMatch) {
            resolve(JSON.parse(jsonMatch[0]))
          } else {
            reject(new Error('AI 返回格式异常'))
          }
        } catch (e) {
          reject(e)
        }
      })
    })

    req.on('error', reject)
    req.write(body)
    req.end()
  })
}
```

> **通义千问 API Key 获取**：
> 1. 访问 [dashscope.console.aliyun.com](https://dashscope.console.aliyun.com)
> 2. 新注册送 **100 万 Token** 免费额度
> 3. 在云函数控制台 → 环境变量中设置 `QWEN_API_KEY=sk-xxx`

## 六、AI 方案

### 6.1 为什么不自训练模型？

Demo 阶段自训练模型成本太高（需要标注数据、GPU 服务器），直接用大模型多模态 API **最简单**：

| 方案 | 准确率 | 成本 | 开发时间 |
|------|-------|------|---------|
| 自训练分类模型 | 较高 | 高（GPU + 数据标注）| 4-8 周 |
| 大模型 API（推荐）| 中等偏上 | 极低（有免费额度）| 1-2 天 |

### 6.2 备选 AI 服务

| 服务 | 免费额度 | 特点 |
|------|---------|------|
| **通义千问 qwen-audio-turbo**（推荐）| 新注册 100 万 Token | 阿里云，支持音频输入 |
| 讯飞开放平台 | 每日 500 次免费 | 国内，音频处理能力强 |
| Kimi（月之暗面）| 新注册有免费额度 | 文字理解好，需先转文字 |

> **MVP 阶段推荐**：通义千问 qwen-audio-turbo，直接输入音频，不需要额外的语音转文字步骤，最简单。

### 6.3 兜底方案

当 AI 接口调用失败或返回格式异常时，使用规则给出默认回复：

```javascript

    suggestion: '陪伴是最好的礼物',
```

## 附：关键问题 FAQ

**Q：通义千问音频 API 对宠物叫声效果怎么样？**
A：大模型对音频的理解能力参差不齐，初期准确率可能 60-70%。这对 Demo 够用，真正的体验优化靠 Prompt 调整。建议在 Prompt 里多描述宠物种类和典型叫声特征。

**Q：云函数 60 秒够用吗？**
A：通义千问音频接口一般 5-15 秒响应，足够。如果偶尔超时，可以在前端加一个「正在努力翻译中...」的等待动画，用户接受度很高。

**Q：朋友圈分享需要体验版还是正式版？**
A：正式版才能无限制转发。体验版只能扫码，适合小范围内测。建议先用体验版给朋友测试反馈，觉得 OK 了再正式上线。

**Q：后续如果用户多了要付费吗？**
A：云开发免费版：日活 ≤ 1000 用户基本不会超额。日活超过 1000 后，按量付费也很便宜（几十块/月）。AI API 费用：通义千问约 0.004元/千 Token，分析一次约消耗 1000 Token，即 0.004 元/次，1 万次分析 = 40 元。
