```javascript
Page({
  data: {
    isRecording: false,

    todayMood: '',

      .where({ date: today }).limit(1).get().catch(() => ({ data: [] }))
    if (res.data[0]) this.setData({ todayMood: res.data[0].moodLabel })
  },

  onRecordStart() {},
  onRecordEnd() {},
```

### 4.1 创建云函数

右键 `cloudfunctions` → 新建 Node.js 云函数 → 命名 `analyzeAudio`

**`cloudfunctions/analyzeAudio/package.json`**：

**`cloudfunctions/analyzeAudio/index.js`**：

```javascript
const cloud = require('wx-server-sdk')
const https = require('https')

cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })

exports.main = async (event) => {
  const { taskId, petName = '宠物', petType = 'cat' } = event
  const petTypeText = petType === 'cat' ? '猫' : '狗'

  try {
    // 1. 从云存储下载音频转 base64
    const fileRes = await cloud.downloadFile({ fileID: taskId })
    const audioBase64 = fileRes.fileContent.toString('base64')

    // 2. 调用通义千问多模态 API
    const aiResult = await callQwenAPI(audioBase64, petName, petTypeText)

    // 3. 分析完成后删除音频文件（节省存储）
    await cloud.deleteFile({ fileList: [taskId] }).catch(() => {})

    // 4. 保存记录到云数据库
    const db = cloud.database()
    await db.collection('records').add({
      data: {
        petName,
        petType,
        ...aiResult,
        createdAt: db.serverDate(),
      }
    })

    return { success: true, ...aiResult }

  } catch (e) {
    console.error('分析失败:', e.message)
    // 兜底返回，不让用户白等
    return {
      success: false,
      emotion: 'happy',
      confidence: 60,
      description: `主人～${petName}想让你多陪陪我！快来摸摸我吧～ 🐾`,
      suggestion: '陪伴是最好的礼物',
      healthAlert: false,
    }
  }
}

function callQwenAPI(audioBase64, petName, petTypeText) {
  return new Promise((resolve, reject) => {
    const prompt = `你是专业的宠物行为分析师。
请分析这段${petTypeText}（叫 ${petName}）的叫声，严格按以下 JSON 格式回复，不要有其他内容：
{
  "emotion": "情绪英文key，从 happy/anxious/angry/fearful/hungry/playful/bored/uncomfortable 中选一个",
  "confidence": 置信度整数0-100,
  "description": "50-60字，第一人称宠物视角，活泼可爱的情绪解读，结尾带emoji",
  "suggestion": "给主人的建议，15字以内",
  "healthAlert": false或true（只有uncomfortable时为true）
}`

    const body = JSON.stringify({
      model: 'qwen-audio-turbo',
      input: {
        messages: [{
          role: 'user',
          content: [
            { audio: `data:audio/mp3;base64,${audioBase64}` },
            { text: prompt }
          ]
        }]
      }
    })

    const req = https.request({
      hostname: 'dashscope.aliyuncs.com',
      path: '/api/v1/services/aigc/multimodal-generation/generation',
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${process.env.QWEN_API_KEY}`,
        'Content-Length': Buffer.byteLength(body),
      }
    }, (res) => {
      let data = ''
      res.on('data', chunk => { data += chunk })
      res.on('end', () => {
        try {
          const resp = JSON.parse(data)
          const text = resp.output?.choices?.[0]?.message?.content?.[0]?.text || ''
          const match = text.match(/\{[\s\S]*\}/)
          if (match) resolve(JSON.parse(match[0]))
          else reject(new Error('AI 返回格式异常: ' + text))
        } catch (e) { reject(e) }
      })
    })
    req.on('error', reject)
    req.write(body)
    req.end()
  })
}
```

```javascript
// pages/checkin/index.js
Page({
  data: {
    petName: '', petType: 'cat',
    selectedMood: 'happy',
    note: '', streakDays: 0,
    moods: [
      { key: 'happy',         emoji: '😄', label: '开心' },
      { key: 'normal',        emoji: '😐', label: '一般' },
      { key: 'sad',           emoji: '😔', label: '难过' },
      { key: 'anxious',       emoji: '😰', label: '焦虑' },
      { key: 'angry',         emoji: '😠', label: '生气' },
      { key: 'uncomfortable', emoji: '🤒', label: '不舒服' },
    ]
  },
  onLoad() {
    this.loadPetInfo()
    this.loadStreak()
  },
  async loadPetInfo() {
    const db = wx.cloud.database()
    const res = await db.collection('pets').limit(1).get().catch(() => ({ data: [] }))
    if (res.data[0]) this.setData({ petName: res.data[0].name, petType: res.data[0].type })
  },
  async loadStreak() {
    // 查连续打卡天数（简单实现：查最近30条按日期判断）
    const db = wx.cloud.database()
    const res = await db.collection('checkins').orderBy('date', 'desc').limit(30).get().catch(() => ({ data: [] }))
    let streak = 0
    const today = new Date()
    for (let i = 0; i < res.data.length; i++) {
      const d = new Date(res.data[i].date)
      const diff = Math.round((today - d) / 86400000)
      if (diff === i) streak++
      else break
    }
    this.setData({ streakDays: streak })
  },
  selectMood(e) { this.setData({ selectedMood: e.currentTarget.dataset.key }) },
  onNoteInput(e) { this.setData({ note: e.detail.value }) },
  async doCheckin() {
    const today = new Date().toISOString().slice(0, 10)
    const moodLabel = this.data.moods.find(m => m.key === this.data.selectedMood)?.label || ''
    const db = wx.cloud.database()
    // 防止重复打卡
    const exist = await db.collection('checkins').where({ date: today }).count()
    if (exist.total > 0) {
      wx.showToast({ title: '今天已经打卡啦～', icon: 'none' })
      return
    }
    await db.collection('checkins').add({
      data: { date: today, mood: this.data.selectedMood, moodLabel, note: this.data.note }
    })
    wx.showToast({ title: '打卡成功 🎉', icon: 'success' })
    setTimeout(() => wx.navigateBack(), 1500)
  },
})
```

```javascript
// pages/profile/index.js
Page({
  data: { pet: null, showEdit: false, form: { name:'', type:'cat', breed:'', birthday:'' } },
  onLoad() { this.loadPet() },
  onShow() { this.loadPet() },
  async loadPet() {
    const db = wx.cloud.database()
    const res = await db.collection('pets').limit(1).get().catch(() => ({ data: [] }))
    this.setData({ pet: res.data[0] || null })
  },
  openEdit() {
    const p = this.data.pet
    this.setData({
      showEdit: true,
      form: p ? { name: p.name, type: p.type, breed: p.breed||'', birthday: p.birthday||'' } : { name:'', type:'cat', breed:'', birthday:'' }
    })
  },
  closeEdit() { this.setData({ showEdit: false }) },
  setType(e) { this.setData({ 'form.type': e.currentTarget.dataset.type }) },
  async savePet() {
    const { form, pet } = this.data
    if (!form.name.trim()) { wx.showToast({ title: '请填写宠物名字', icon: 'none' }); return }
    const db = wx.cloud.database()
    if (pet) {
      await db.collection('pets').doc(pet._id).update({ data: form })
    } else {
      await db.collection('pets').add({ data: form })
    }
    this.setData({ showEdit: false })
    this.loadPet()
    wx.showToast({ title: '保存成功', icon: 'success' })
  },
})
```

## 遇到问题怎么办

| 问题 | 解决方向 |
|------|---------|
| 云函数超时 | 通义千问响应一般 5-10s，在 60s 限制内，偶发超时属正常，前端加 loading 提示 |
| AI 返回格式不对 | 检查 Prompt，用 `try/catch` + 兜底数据保护 |
| 小程序审核拒绝 | 最常见原因：无隐私政策页、功能描述不清晰，按拒绝理由修改再提 |
| 录音在真机没声音 | 真机调试时需要用「真机调试」模式，不能用模拟器测试录音 |
| 云数据库权限报错 | 检查集合权限是否设置为「仅创建者可读写」 |
